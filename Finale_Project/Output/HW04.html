Summary <- Data_FL %>%
  summarize(across(c(TotalPopul : PercentP_7),
                   list( mu = ~ mean (.x, na.rm = TRUE),
                         sd = ~ sd (.x, na.rm = TRUE),
                         min = ~ min (.x, na.rm = TRUE),
                         max = ~ max (.x, na.rm = TRUE))))
Plot1 <- Summary %>%
ggplot() +
  aes(y = "TotalPopul", x = "Unemployme") +
  geom_jitter(alpha = .5) +
